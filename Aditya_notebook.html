df['average_score'] = df[['math score', 'reading score', 'writing score']].mean(axis=1)
def performance_category(score):
    if score < 50:
        return "Low"
    elif score < 70:
        return "Medium"
    else:
        return "High"

df['performance_category'] = df['average_score'].apply(performance_category)
